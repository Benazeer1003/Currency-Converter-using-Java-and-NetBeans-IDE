    private void convertActionPerformed(java.awt.event.ActionEvent evt) {                                        
        double amount;
        String fromcurrency;
        String tocurrency;
        double convertAmount;
        try{
            amount = Double.parseDouble(amounttxtfield.getText());
        }
        catch(NumberFormatException e){
        JOptionPane.showMessageDialog(this,"Please enter a valid number","Exception",JOptionPane.ERROR_MESSAGE);
        amounttxtfield.setText("");
        amounttxtfield.requestFocus();
        return;
        }
        fromcurrency = fromcurrencycb.getSelectedItem().toString().substring(0,3);
        tocurrency = tocurrencycb.getSelectedItem().toString().substring(0,3);
        convertAmount = convertCurrency(amount,fromcurrency,tocurrency);
        amountconvertedtf.setText(amount + " " + fromcurrency+ " = " + convertAmount + " " + tocurrency);
    }                                       

    private void resetActionPerformed(java.awt.event.ActionEvent evt) {                                      
        amounttxtfield.setText("");
        amountconvertedtf.setText("");
        fromcurrencycb.setSelectedIndex(-1);
        tocurrencycb.setSelectedIndex(-1);
    }                                     

     private double convertCurrency(double amount, String fromcurrency, String tocurrency) {
        Map<String, Double> exchangeRates = new HashMap<>();
        exchangeRates.put("GBP", 1.0);
        exchangeRates.put("CAD", 1.77);
        exchangeRates.put("NGN", 2047.81);
        exchangeRates.put("MXN", 24.74);
        exchangeRates.put("EUR", 1.17);
        exchangeRates.put("CHF", 1.11);
        exchangeRates.put("AUD", 1.93);
        exchangeRates.put("CNY", 9.29);
        exchangeRates.put("INR", 109.28);
        exchangeRates.put("USD", 1.30);
        exchangeRates.put("ZMK", 33.44);
     if (!exchangeRates.containsKey(fromcurrency) || !exchangeRates.containsKey(tocurrency)) {
            throw new IllegalArgumentException("Invalid currency code");
        }

        double fromRate = exchangeRates.get(fromcurrency);
        double toRate = exchangeRates.get(tocurrency);

        return amount * (toRate / fromRate);}
